/*テーブル作成*/
CREATE TABLE event (
event_id INT AUTO_INCREMENT PRIMARY KEY,
store_id INT NOT NULL,
event_date DATE NOT NULL,
event_name VARCHAR(100) NOT NULL,
event_remark VARCHAR(100) NOT NULL
) ;
