/*テーブル削除*/
DROP TABLE orderlist;

/*テーブル作成*/
CREATE TABLE orderlist(
order_id INT AUTO_INCREMENT PRIMARY KEY,
commodity_id INT NOT NULL,
order_datetime DATE NOT NULL,
order_quantity INT NOT NULL
) ;
